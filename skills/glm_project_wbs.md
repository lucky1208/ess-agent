### 4.2 风险应对代码

```python
def risk_assessment(risks: list):
    """风险评估与排序"""
    impact_score = {"极高": 5, "高": 4, "中": 3, "低": 2, "极低": 1}
    prob_score = {"极高": 5, "高": 4, "中": 3, "低": 2, "极低": 1}
    
    assessed = []
    for r in risks:
        p = prob_score.get(r["概率"], 3)
        i = impact_score.get(r["影响"], 3)
        score = p * i
        if score >= 16:
            level = "极高"
        elif score >= 9:
            level = "高"
        elif score >= 4:
            level = "中"
        else:
            level = "低"
        
        assessed.append({
            "编号": r["编号"],
            "描述": r["描述"],
            "概率": r["概率"],
            "影响": r["影响"],
            "风险得分": score,
            "风险等级": level,
            "应对": r["应对"]
        })
    
    assessed.sort(key=lambda x: x["风险得分"], reverse=True)
    return assessed
```

### 6.2 费用预算代码

```python
def project_budget(
    capacity_mwh: float,
    power_mw: float,
    battery_price_per_wh: float = 0.40,
    pcs_price_per_kw: float = 200,
    liquid_cooling: bool = True
):
    """项目费用预算"""
    
    # 设备费
    battery_cost = capacity_mwh * 1000 * battery_price_per_wh
    pcs_cost = power_mw * 1000 * pcs_price_per_kw / 10000
    bms_cost = capacity_mwh * 5
    transformer_cost = power_mw * 9
    switchgear_cost = power_mw * 16
    cable_cost = power_mwh * 6
    protection_cost = power_mw * 4
    comm_cost = power_mw * 3
    thermal_cost = capacity_mwh * (8 if liquid_cooling else 6)
    fire_cost = capacity_mwh * 5
    meter_cost = power_mw * 2
    ems_cost = 30
    container_cost = (capacity_mwh / 2.5) * 25
    
    equipment_total = (battery_cost + pcs_cost + bms_cost + transformer_cost +
                      switchgear_cost + cable_cost + protection_cost + comm_cost +
                      thermal_cost + fire_cost + meter_cost + ems_cost + container_cost)
    
    # 建安费(约设备费的25~30%)
    civil_cost = capacity_mwh * 12
    install_cost = power_mw * 16
    cable_lay_cost = power_mwh * 6
    commission_cost = power_mw * 10
    construction_total = civil_cost + install_cost + cable_lay_cost + commission_cost
    
    # 其他费
    design_cost = equipment_total * 0.06
    supervision_cost = construction_total * 0.10
    compliance_cost = 30
    insurance_cost = equipment_total * 0.01
    other_total = design_cost + supervision_cost + compliance_cost + insurance_cost
    
    # 预备费
    contingency = (equipment_total + construction_total + other_total) * 0.05
    
    # 建设期利息(假设70%融资, 利率4.5%, 工期6月)
    financing = (equipment_total + construction_total) * 0.7 * 0.045 * 0.5
    
    total = equipment_total + construction_total + other_total + contingency + financing
    
    return {
        "一、设备费": {
            "电池系统": round(battery_cost, 0),
            "PCS": round(pcs_cost, 0),
            "BMS": round(bms_cost, 0),
            "变压器": round(transformer_cost, 0),
            "开关柜": round(switchgear_cost, 0),
            "电缆": round(cable_cost, 0),
            "保护装置": round(protection_cost, 0),
            "通信设备": round(comm_cost, 0),
            "温控系统": round(thermal_cost, 0),
            "消防系统": round(fire_cost, 0),
            "电能表": round(meter_cost, 0),
            "EMS系统": round(ems_cost, 0),
            "集装箱": round(container_cost, 0),
            "小计": round(equipment_total, 0)
        },
        "二、建安费": {
            "土建": round(civil_cost, 0),
            "安装": round(install_cost, 0),
            "电缆敷设": round(cable_lay_cost, 0),
            "调试": round(commission_cost, 0),
            "小计": round(construction_total, 0)
        },
        "三、其他费": round(other_total, 0),
        "四、预备费": round(contingency, 0),
        "五、建设期利息": round(financing, 0),
        "总投资(万元)": round(total, 0),
        "单位投资(元/Wh)": round(total / (capacity_mwh * 1000), 2)
    }

# 示例
budget = project_budget(capacity_mwh=10, power_mw=5)
import json
print(json.dumps(budget, ensure_ascii=False, indent=2))
```
